```c
#include <stdio.h> 

int main() {
    int x, y;

    printf("Welche number x: ");
    scanf_s("%d", &x);

    if (x < 0) {
        if (x <= -10) {
            y = -3 * x / 7 + 10;
            printf("y = -3x/7 + 10 = %d\n", y);
        } else if (x > -10 ) {
            printf("kein antwort\n");
        }
    }
    else if (x >= 0) {
        if (x < 7) {
            y = -6 * x * x + 8;
            printf("y = -6x^2 + 8 = %d\n", y);
        }else if (x >= 7) {
            if (x < 11) {
                printf("kein antwort\n");
            } else if (x >= 11) {
                y = -3 * x / 7 + 10;
                printf("y = -3x/7 + 10 = %d\n", y);
            }
        }
        return 0;
    }
}
```
