#без булевых операторов 
```c
#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h> //библиотека С

int main() {
    int x, y;

    printf("Welche number x: "); 
    scanf_s("%d", &x); 

    //цикл if-else
    if (x >= 0){
        if  (x < 7) { //and
            y = -6 * x * x + 8;
            printf("y = -6x^2 + 8 = %d\n", y);
        } else if (x >= 11) { //or
            y = -3 * x / 7 + 10;
            printf("y = -3x/7 + 10 = %d\n", y);
        } else {
            printf("kein antwort\n");
        }
    } else {
        if (x <= -10) { //or
            y = -3 * x / 7 + 10;
            printf("y = -3x/7 + 10 = %d\n", y);
        } else {
            printf("kein antwort\n");
        }
    }
    return 0;
}
```
